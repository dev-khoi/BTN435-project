## LINUX static routing set up
# Router interfaces:
eth is a naming convention for (NIC) in Linux
    eth0 -> Lighting subnet   192.168.10.1/26
    eth1 -> Thermostat subnet 192.168.10.65/27
    eth2 -> Security subnet   192.168.10.97/28

# On server

# Networks will be directly connected once interfaces are configured.
# interface setup:
ip addr add 192.168.10.1/26 dev eth0
ip addr add 192.168.10.65/27 dev eth1
ip addr add 192.168.10.97/28 dev eth2
ip link set eth0 up
ip link set eth1 up
ip link set eth2 up
sysctl -w net.ipv4.ip_forward=1

####################
# On lighting host #
####################
ip route add 192.168.10.64/27 via 192.168.10.1
ip route add 192.168.10.96/28 via 192.168.10.1

######################
# On thermostat host #
######################
ip route add 192.168.10.0/26 via 192.168.10.65
ip route add 192.168.10.96/28 via 192.168.10.65

####################
# On security host #
####################
ip route add 192.168.10.0/26 via 192.168.10.97
ip route add 192.168.10.64/27 via 192.168.10.97

# Optional default route style (per subnet host):
# ip route add default via <subnet-gateway>
